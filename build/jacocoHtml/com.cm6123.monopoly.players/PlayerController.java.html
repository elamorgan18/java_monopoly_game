package com.cm6123.monopoly.players;
/**
 * A class that manages the players.
 */
public class PlayerController {

    /**
     * A method to check the number of players is an integer and is between 2 and 10.
     *
     * @param numberOfPlayers - The number of players the user wants to play with.
     * @return a boolean to see if number is valid
     */
    public boolean validateNumberOfPlayers(final String numberOfPlayers) {

        boolean isValid = false;
        int playerCount = 0;

        // Converting the input to an int.
        try {
            if (numberOfPlayers != null) {
                playerCount = getNumberOfPlayers(numberOfPlayers);
                isValid = true;
            }
        } catch (NumberFormatException e) {
            System.out.println("Number of players must be an integer. Please try again.");
            isValid = false;
        }

        // Checking if the input is between 2 and 10.
        if (isValid) {
            if (playerCount < 2 || playerCount > 10) {
                isValid = false;
                System.out.println("Number of players must be between 2 and 10. Please try again.");
            }
        }
        return isValid;

    }

    /**
     * A method to retrieve the number of players.
     * @param numberOfPlayers the number of players the user inputted as a string.
     * @return an integer of the number of players.
     */
    public int getNumberOfPlayers(final String numberOfPlayers) {
        return Integer.parseInt(numberOfPlayers.trim());
    }
}
